        \\
        \\

            \\
            \\
            = \left[(\dfrac{\partial a_z}{\partial y}-\dfrac{\partial a_y}{\partial z})\hat{i}-(\dfrac{\partial a_z}{\partial x}-\dfrac{\partial a_x}{\partial z})\hat{j}+(\dfrac{\partial a_y}{\partial x}-\dfrac{\partial a_x}{\partial y})\hat{k}\right] \\ \\
            \\ 
            +\left[(\dfrac{\partial b_z}{\partial y}-\dfrac{\partial b_y}{\partial z})\hat{i}-(\dfrac{\partial b_z}{\partial x}-\dfrac{\partial b_x}{\partial z})\hat{j}+(\dfrac{\partial b_y}{\partial x}-\dfrac{\partial b_x}{\partial y})\hat{k}\right] \\
            \\
            \\
            =\begin{vmatrix}
              \hat{i} & \hat{j} & \hat{k} \\
              \\
              \dfrac{\partial}{\partial x} & \dfrac{\partial}{\partial y} & \dfrac{\partial}{\partial z} \\
              \\
              a_x & a_y & a_z
            \end{vmatrix} + \begin{vmatrix}
              \hat{i} & \hat{j} & \hat{k} \\
              \\
              \dfrac{\partial}{\partial x} & \dfrac{\partial}{\partial y} & \dfrac{\partial}{\partial z} \\
              \\
              b_x & b_y & b_z
            \end{vmatrix} \\
            \\
            \\
            \\
            \therefore ~~ \overrightarrow{\nabla} \times (\mathbf{a}+\mathbf{b})=\overrightarrow{\nabla} \times \mathbf{a}+\overrightarrow{\nabla} \times \mathbf{b}
          $
        \end{enumerate}
      }
    
    \item What is the operator $\mathbf{V}\cdot \mathbf{\nabla}$, where $\mathbf{V}$ is a generic vector field? [Hint: you can find this discussed in the textbook.] Prove  that $(\mathbf{V}\cdot \mathbf{\nabla})\mathbf{r}=\mathbf{V}$, where $\mathbf{r}$ is the position vector. [Note: in this class we always use the convention that an operator act on what is on its \emph{right}.]
    
      \textcolor{hwColor}{
        $\mathbf{V}.\overrightarrow{\nabla}$ is ready to act on a function $f(x,y,z)$. In other words, $\mathbf{V}.\overrightarrow{\nabla}$
        is an operator that maps a vector valued function to a scalar. \\
        This is a scalar operator, but it obviously can be applied to a scalar field, resulting in a scalar field, or to a vector field resulting in a vector field: \\
        \\
        $\mathbf{V}.\overrightarrow{\nabla} \equiv \left[\mathbf{V}_x \dfrac{\partial}{\partial x}+\mathbf{V}_y \dfrac{\partial}{\partial y}+\mathbf{V}_z \dfrac{\partial}{\partial z}\right]$ \\
        \\
        $
          (\mathbf{V}.\overrightarrow{\nabla})\mathbf{r}=\mathbf{V} \\
          \\
          L.H.S: \\ \\
          \left(\mathbf{V}_x \dfrac{\partial}{\partial x}+\mathbf{V}_y \dfrac{\partial}{\partial y}+\mathbf{V}_z \dfrac{\partial}{\partial z}\right)(x\hat{i}+y\hat{j}+z\hat{k}) \\
          \\
          =\left(\mathbf{V}_x \dfrac{\partial}{\partial x}+\mathbf{V}_y \dfrac{\partial}{\partial y}+\mathbf{V}_z \dfrac{\partial}{\partial z}\right) x\hat{i} \\
          \\
          +\left(\mathbf{V}_x \dfrac{\partial}{\partial x}+\mathbf{V}_y \dfrac{\partial}{\partial y}+\mathbf{V}_z \dfrac{\partial}{\partial z}\right) y\hat{j} \\
          \\
          +\left(\mathbf{V}_x \dfrac{\partial}{\partial x}+\mathbf{V}_y \dfrac{\partial}{\partial y}+\mathbf{V}_z \dfrac{\partial}{\partial z}\right) z\hat{k} \\
          \\
          \\
          \\
          =\left(\mathbf{V}_x \dfrac{\partial x}{\partial x}+\mathbf{V}_y \dfrac{\partial x}{\partial y}+\mathbf{V}_z \dfrac{\partial x}{\partial z}\right) \hat{i} \\
          \\
          +\left(\mathbf{V}_x \dfrac{\partial y}{\partial x}+\mathbf{V}_y \dfrac{\partial y}{\partial y}+\mathbf{V}_z \dfrac{\partial y}{\partial z}\right) \hat{j} \\
          \\
          +\left(\mathbf{V}_x \dfrac{\partial z}{\partial x}+\mathbf{V}_y \dfrac{\partial z}{\partial y}+\mathbf{V}_z \dfrac{\partial z}{\partial z}\right) \hat{k} \\ 
          \\
          \\
          \\
          =\left(\mathbf{V}_x \dfrac{\partial x}{\partial x}\right) \hat{i}+\left(\mathbf{V}_y \dfrac{\partial y}{\partial y}\right) \hat{j}+\left(\mathbf{V}_z \dfrac{\partial z}{\partial z}\right) \hat{k}=\mathbf{V}_x \hat{i}+\mathbf{V}_y\hat{j}+\mathbf{V}_z \hat{k}=\mathbf{V} \\
          \\
          \\
          \therefore (\mathbf{V}.\overrightarrow{\nabla})\mathbf{r}=\mathbf{V}
        $
      }
